package com.gubee.stockreconciliation.infrastructure.adapter.out.persistence;

import com.gubee.stockreconciliation.application.port.out.StockEventRepositoryPort;
import com.gubee.stockreconciliation.domain.model.EventStatus;
import com.gubee.stockreconciliation.domain.model.EventType;
import com.gubee.stockreconciliation.domain.model.StockEvent;
import com.gubee.stockreconciliation.infrastructure.adapter.out.persistence.repository.StockEventJpaRepository;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Component;

import java.util.List;
import java.util.Optional;

@Component
@RequiredArgsConstructor
public class StockEventPersistenceAdapter implements StockEventRepositoryPort {

    private final StockEventJpaRepository jpaRepository;

    @Override
    public boolean existsByEventId(String eventId) {
        return jpaRepository.existsByEventId(eventId);
    }

    @Override
    public Optional<StockEvent> findByEventId(String eventId) {
        return jpaRepository.findByEventId(eventId);
    }

    @Override
    public List<StockEvent> findByStatus(EventStatus status) {
        return jpaRepository.findByStatus(status);
    }

    @Override
    public List<StockEvent> findPendingCancellations(String marketplace, String accountId,
                                                      String externalOrderId, String sku) {
        return jpaRepository.findPendingCancellations(
            EventType.ORDER_CANCELLED, EventStatus.PENDING,
            marketplace, accountId, externalOrderId, sku);
    }

    @Override
    public StockEvent save(StockEvent event) {
        return jpaRepository.save(event);
    }
}
